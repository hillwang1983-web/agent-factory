> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deploy a minimal Hermes-based AI Agent factory in `/Users/hill/open5gs` so the team can immediately start debugging one MVP 5G core-network ADU through context, contract, test, development, debug, and evidence stages.

**Tech Stack:** Hermes CLI, Python 3 standard library, JSON registries, Markdown prompt templates, local shell/CI commands, Open5GS workspace files.

## File Structure

| File | Responsibility |
| --- | --- |
| `.ai-agent/registry/agents.json` | Defines MVP Agent roles, prompt paths, Hermes arguments, model override, and whether worktree isolation is required. |
| `.ai-agent/registry/adu.json` | Defines the MVP ADU, current state, allowed paths, validation commands, evidence requirements, retry count, and next-agent flow. |
| `.ai-agent/registry/runs.json` | Append-only run index generated by wrapper scripts. |
| `.ai-agent/prompts/*.md` | Role instructions for each Hermes-executed Agent. |
| `.ai-agent/context-packs/common.md` | Shared context for all MVP Agents. |
| `.ai-agent/contracts/REQ-MVP-001.json` | MVP contract and output requirements for the first ADU. |
| `scripts/hermes_agent_run.py` | Runs one Agent for one ADU through Hermes `--oneshot`, stores output, validates structured result, and updates state. |
| `scripts/hermes_agent_next.py` | Selects the next runnable ADU and Agent based on the ADU state machine. |
| `scripts/hermes_agent_status.py` | Prints current ADU state, last runs, retry counts, and next action. |

## State Machine

MVP state flow:

```text
created
  -> contexted
  -> contracted
  -> test_red
  -> implemented
  -> debugged
  -> evidenced
  -> mvp_ready
```

State to Agent mapping:

| State | Next Agent |
| --- | --- |
| `created` | `context-pack` |
| `contexted` | `contract` |
| `contracted` | `testwriter` |
| `test_red` | `developer` |
| `implemented` | `buildfix-debugger` |
| `debugged` | `evidence` |
| `evidenced` | none |

Failure rule:

```text
If the same ADU fails 3 consecutive runs, set state to human_gate.
```

## Task 8: Create Hermes Runner Script

**Files:**

- [ ] **Step 1: Create `scripts/hermes_agent_run.py`**

Write this exact file:

```python
#!/usr/bin/env python3
import argparse
import datetime as dt
import json
import os
import re
import subprocess
import sys
from pathlib import Path


ROOT = Path(__file__).resolve().parents[1]
REGISTRY = ROOT / ".ai-agent" / "registry"


STATE_NEXT = {
    "created": ("context-pack", "contexted"),
    "contexted": ("contract", "contracted"),
    "contracted": ("testwriter", "test_red"),
    "test_red": ("developer", "implemented"),
    "implemented": ("buildfix-debugger", "debugged"),
    "debugged": ("evidence", "evidenced"),
}


def load_json(path):
    with path.open("r", encoding="utf-8") as f:
        return json.load(f)


def save_json(path, data):
    tmp = path.with_suffix(path.suffix + ".tmp")
    with tmp.open("w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
        f.write("\n")
    tmp.replace(path)


def find_adu(adu_data, adu_id):
    for adu in adu_data["adus"]:
        if adu["id"] == adu_id:
            return adu
    raise SystemExit(f"ADU not found: {adu_id}")


def render_prompt(prompt_text, adu, agent_name):
    common_path = ROOT / ".ai-agent" / "context-packs" / "common.md"
    common = common_path.read_text(encoding="utf-8") if common_path.exists() else ""
    payload = {
        "agent": agent_name,
        "adu": adu,
        "common_context": common,
    }
    rendered = prompt_text.replace("{{ADU_ID}}", adu["id"])
    rendered += "\n\n# Runtime Payload\n\n"
    rendered += "```json\n"
    rendered += json.dumps(payload, ensure_ascii=False, indent=2)
    rendered += "\n```\n"
    return rendered


def extract_json_result(text):
    blocks = re.findall(r"```json\s*(\{.*?\})\s*```", text, flags=re.DOTALL)
    if not blocks:
        return None
    for block in reversed(blocks):
        try:
            return json.loads(block)
        except json.JSONDecodeError:
            continue
    return None


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--adu", required=True)
    parser.add_argument("--agent", required=True)
    parser.add_argument("--dry-run", action="store_true")
    args = parser.parse_args()

    agents = load_json(REGISTRY / "agents.json")
    adu_data = load_json(REGISTRY / "adu.json")
    runs = load_json(REGISTRY / "runs.json")

    adu = find_adu(adu_data, args.adu)
    agent_cfg = agents["agents"].get(args.agent)
    if not agent_cfg:
        raise SystemExit(f"Agent not found: {args.agent}")

    prompt_path = ROOT / agent_cfg["prompt"]
    prompt_text = prompt_path.read_text(encoding="utf-8")
    prompt = render_prompt(prompt_text, adu, args.agent)

    timestamp = dt.datetime.now().strftime("%Y%m%d-%H%M%S")
    run_dir = ROOT / ".ai-agent" / "runs" / f"{timestamp}-{adu['id']}-{args.agent}"
    run_dir.mkdir(parents=True, exist_ok=True)
    (run_dir / "prompt.md").write_text(prompt, encoding="utf-8")

    cmd = [agents.get("hermes_bin", "hermes")]
    cmd.extend(agent_cfg.get("hermes_args", []))
    cmd.extend(["-z", prompt])

    if args.dry_run:
        print(" ".join(cmd[: len(cmd) - 1]) + " <prompt>")
        return

    proc = subprocess.run(
        cmd,
        cwd=str(ROOT),
        text=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
    )

    (run_dir / "stdout.md").write_text(proc.stdout, encoding="utf-8")
    (run_dir / "stderr.md").write_text(proc.stderr, encoding="utf-8")

    result = extract_json_result(proc.stdout)
    if proc.returncode != 0:
        adu["retry_count"] = int(adu.get("retry_count", 0)) + 1
        run_result = "failed"
    elif result is None:
        adu["retry_count"] = int(adu.get("retry_count", 0)) + 1
        run_result = "unstructured"
    else:
        run_result = result.get("result", "unknown")
        next_state = result.get("next_state")
        if next_state:
            adu["state"] = next_state
        adu["retry_count"] = 0
        for artifact in result.get("artifacts", []):
            if artifact not in adu["artifacts"]:
                adu["artifacts"].append(artifact)

    if int(adu.get("retry_count", 0)) >= int(adu.get("max_retries", 3)):
        adu["state"] = "human_gate"
        adu["human_gate_required"] = True

    run_record = {
        "timestamp": timestamp,
        "adu_id": adu["id"],
        "agent": args.agent,
        "returncode": proc.returncode,
        "result": run_result,
        "run_dir": str(run_dir.relative_to(ROOT)),
        "parsed_result": result,
    }
    runs["runs"].append(run_record)

## Deployment Note: JSON Extraction Regex

The runner must use a single backslash inside the Python raw-string regex:

```python
blocks = re.findall(r"```json\s*(\{.*?\})\s*```", text, flags=re.DOTALL)
```

Do not use this double-escaped version in a Python raw string:

```python
blocks = re.findall(r"```json\\s*(\\{.*?\\})\\s*```", text, flags=re.DOTALL)
```

The double-escaped version matches literal backslashes and causes Hermes Agent JSON output to be classified as `unstructured`.
